  - hide TaskList component
  - hide hamburger on screen < 1024

  TaskList Classes:    is-hidden: true
  Burger Classes:      is-active: false, is-hidden: true

*************************************************

ProjectList Classes:  
    CONTROL                  -> on navActive, on !haveProjects
    SET ''                   -> on hamburger click, on no projects

    CONTROL                  -> on !navActive, on haveProjects, 
    SET is-2 is-hidden-touch -> on x click, on add project, on project click

*************************************************

TaskList Class is-hidden: 
    CONTROL   -> !haveProjects OR navActive
    SET TRUE  -> on hamburger click, on no projects
    SET FALSE -> on x click, on project click, on add project

*************************************************

Burger Class is-hidden: 
    CONTROL   -> !haveProjects
    SET TRUE  -> on no projects
    SET FALSE -> on add project
             
Burger Class is-active: 
    CONTROL   -> navActive
    SET TRUE  -> on hamburger click
    SET FALSE -> on x click, on project click
             
*************************************************

hamburger click -> set navActive true, set ProjectList class ''
x click         -> set navActive false, set ProjectList class is-2 is-hidden-touch
project click   -> set navActive false, set ProjectList class is-2 is-hidden-touch
